Q1: Write a python program using mean end analysis algorithmproblem of transforming a string of lowercase
letters into another string.
from collections import deque
def transform(start,goal):
q=deque([(start,[start])]); visited={start}
while q:
s,path=q.popleft()
if s==goal: return path
for i in range(len(s)):
for c in 'abcdefghijklmnopqrstuvwxyz':
ns=s[:i]+c+s[i+1:]
if ns not in visited:
visited.add(ns); q.append((ns,path+[ns]))
return None
