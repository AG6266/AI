Q1: Write a Program to Implement Alpha-Beta Pruning using Python
import math
def alphabeta(node,depth,alpha,beta,maximizing):
if depth==0 or isinstance(node,int): return node
if maximizing:
v=-math.inf
for child in node:
v=max(v,alphabeta(child,depth-1,alpha,beta,False))
alpha=max(alpha,v)
if alpha>=beta: break
return v
else:
v=math.inf
for child in node:
v=min(v,alphabeta(child,depth-1,alpha,beta,True))
beta=min(beta,v)
if alpha>=beta: break
return v
print(alphabeta([[3,5],[6,9]],2,-math.inf,math.inf,True))
